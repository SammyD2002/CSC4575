\node[state]  (bval) [below=of bsign] {Bob validates Alice's X25519 public key with his signature};

\node[state] (bshs) [below=of bval] {Bob uses Alice's validated X25519 public key and his own X25519 private key to obtain the shared secret $Z$.};
